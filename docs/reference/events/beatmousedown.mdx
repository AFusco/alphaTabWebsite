## Description
This event is fired whenever a the user presses the mouse button on a beat.

<EventDescription metadata={metadata} />

## Types

<TypeTable>
    <TypeRow type="net" name="Action<AlphaTab.Model.Beat>" />    
    <TypeRow type="js" name="function(beat, originalEvent)" />    
</TypeTable>

## Parameters

<ParameterTable>
    <ParameterRow platform="all" name="beat" type="AlphaTab.Model.Beat">
        The new beat that is now being played..
    </ParameterRow>
    <ParameterRow platform="jquery" name="originalEvent" type="MouseEvent">
        The original mouse event that lead to trigger of the beat event. For the DOM event it is stored in the event.originalEvent.
    </ParameterRow>
</ParameterTable>

const api = new alphaTab.platform.javaScript.AlphaTabApi(document.querySelector('#alphaTab'));
api.beatMouseDown.on((beat) => {
    startSelectionOnBeat(beat);
});

<TabItem value="jq">

```js
$('#alphaTab').on('alphaTab.beatMouseDown', (e, beat) => {
    originalEvent.preventDefault();
    startSelectionOnBeat(beat);
});
```

</TabItem>
<TabItem value="html">

```js
document.querySelector('#alphaTab').addEventListener('alphaTab.beatMouseDown', (e) => {
    const beat = e.detail;
    e.originalEvent.preventDefault();
    startSelectionOnBeat(beat);
}, false);
```

</TabItem>

var api = new AlphaTabApi<MyControl>(...);
